  final String eventType;
  final DateTime date;
  final String userId;

  Map<String, dynamic> toMap() {
    return {
      'id': id,
      'eventType': eventType,
      'date': date.toIso8601String(),
      'userId': userId,
    };
  }

  factory Booking.fromMap(Map<String, dynamic> map) {
    return Booking(
      id: map['id'],
      eventType: map['eventType'],
      date: DateTime.parse(map['date']),
      userId: map['userId'],
    );
  }
}
